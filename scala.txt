val archived_count_query =
  s"""
     |SELECT alert_code, COUNT(*) AS archived_cnt
     |FROM alert_load_archive
     |GROUP BY alert_code
  """.stripMargin


val rs_arch = stmt.executeQuery(archived_count_query)

var archived_count: Map[String, Int] = Map()


while (rs_arch.next()) {
  val alertCode = rs_arch.getString("alert_code")
  val count = rs_arch.getInt("archived_cnt")
  archived_count += (alertCode -> count)
}
scn_count.foreach { case (alertCode, scnCnt) =>
  val archivedCnt = archived_count.getOrElse(alertCode, 0)
  successAcc.add((alertCode, masterFile1CSVcount, scnCnt, archivedCnt))
}
